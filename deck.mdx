// SignUpScreen.test.tsx

test('User cannot SignUp with wrong credentials', async () => {
  mockSignUpApi.mockImplementationOnce(() =>
    Promise.resolve({
      status: 401,
      data: {
        error: { 'Invalid username or password' }
      }
    }),
  );

  const screen = renderSignUpScreen();

  fireEvent.changeText(screen.getByPlaceholder(/Username/i), 'user@callstack.com');
  fireEvent.changeText(screen.getByPlaceholder(/Password/i), 'Incorrect Password');
  fireEvent.press(screen.getByText('SignUp'));

  expect(mockSignUpApi).toHaveBeenCalledTimes(1);
  expect(mockSignUpApi).toHaveBeenCalledWith({
    username: 'user@callstack.com',
    password: 'Incorrect Password',
  });
  expect(await screen.findByText('Invalid username or password')).toBeTruthy();
});
